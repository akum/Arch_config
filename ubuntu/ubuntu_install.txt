# Install Ubuntu 16.04 Desktop

# Install Atom
wget https://github.com/atom/atom/releases/download/v1.20.1/atom-amd64.deb
sudo apt-get install -y ./atom-amd64.deb

# Install Gitkraken
wget https://release.gitkraken.com/linux/gitkraken-amd64.deb
sudo dpkg -i gitkraken-amd64.deb

# Install i3wm gaps
sudo add-apt-repository -y ppa:simon-monette/i3-gaps
sudo apt-get update
sudo apt-get install -y i3 i3blocks-gaps

# Install extra packages
sudo apt install -y rofi compton fonts-font-awesome cmus notification-daemon feh git

# urxvt
sudo apt-get install -y rxvt-unicode-256color xsel xclip
sudo wget https://raw.githubusercontent.com/muennich/urxvt-perls/master/deprecated/clipboard -O /usr/lib/urxvt/perl/clipboard
sudo wget https://raw.githubusercontent.com/shyiko/urxvt-clipboard-paste-on-right-click/master/clipboard-paste-on-right-click -O /usr/lib/urxvt/perl/clipboard-paste-on-right-click

# Get config files
wget https://raw.githubusercontent.com/s-monette/Arch_config/master/ubuntu/.Xresources -O ~/.Xresources
wget https://raw.githubusercontent.com/s-monette/Arch_config/master/.i3blocks.conf -O ~/.i3blocks.conf
wget https://raw.githubusercontent.com/s-monette/Arch_config/master/.config/compton.conf -O ~/.config/compton.conf
curl https://raw.githubusercontent.com/s-monette/Arch_config/master/ubuntu/.config/i3/config --create-dirs -o ~/.config/i3/config

# i3block file
sudo wget https://raw.githubusercontent.com/s-monette/Arch_config/master/usr/lib/i3blocks/vol -P /usr/lib/i3blocks
sudo wget https://raw.githubusercontent.com/s-monette/Arch_config/master/usr/lib/i3blocks/cmus -P /usr/lib/i3blocks
sudo wget https://raw.githubusercontent.com/s-monette/Arch_config/master/usr/lib/i3blocks/simplecal -P /usr/lib/i3blocks
sudo chmod +x /usr/lib/i3blocks/*

# Theme files
sudo git clone https://github.com/chriskempson/base16-xresources.git /usr/share/base16/
sudo wget https://wallpapers.wallhaven.cc/wallpapers/full/wallhaven-2936.png -P /usr/share/pixmaps/
wget https://raw.githubusercontent.com/s-monette/Arch_config/master/ubuntu/.gtkrc-2.0 -O ~/.gtkrc-2.0
